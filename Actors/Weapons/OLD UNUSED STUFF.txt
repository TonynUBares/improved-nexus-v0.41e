		TNT1 A 0 A_FireBullets(0,0,1,0,"NoPuff") // to use ammo
		TNT1 A 0 A_GunFlash

		UNMK E 10
		UNMK F 10 A_ReFire

		Goto Ready

		UNMK D 40 A_PlaySound("LegUnmaker/incant",0,1.0,FALSE)
		TNT1 A 0 A_Playsound("LegUnmaker/altfire",0,1.0,FALSE)
