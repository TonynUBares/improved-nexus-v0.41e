Actor LegendaryC4Giver : CustomInventory
{
  Scale 0.9
  Inventory.Amount 1
  Inventory.MaxAmount 0
  Inventory.InterHubAmount 0
   Inventory.PickupSound "misc/w_pkup"
   Inventory.PickupMessage "You got the ultimate Legendary C4! straight out of DE_DUST!"
  Tag "Legendary C4"
  States
  {
  Spawn:
	PUC4 A 0 NoDelay ACS_NamedExecuteWithResult("WeaponListCheck",1,1)
  SpawnLoop:
	PUC4 A -1
	Loop
  Pickup:
	TNT1 A 0 A_JumpIfInventory("Legendary C4",1,"JustAmmo")
  All:
	TNT1 A 0 A_GiveInventory("Legendary C4",1)
	TNT1 A 0 A_JumpIfInventory("LegendaryC4GiverCheck",1,"HoldAndDestroy")
	TNT1 A 0 A_GiveInventory("LegendaryC4GiverCheck",1)
	TNT1 A 0 A_GiveInventory("LegendaryC4Ammo",40)
	Stop
  JustAmmo:
    TNT1 A 0 A_JumpIfInventory("LegendaryC4GiverCheck",1,"Null")
	TNT1 A 0 A_GiveInventory("LegendaryC4GiverCheck",1)
	TNT1 A 0 A_GiveInventory("LegendaryC4Ammo",40)
	TNT1 A 0 A_SpawnItemEx("Legendary C4")
	Stop
  }
}

ACTOR "Legendary C4" : Weapon //RUSH B RUSH B!!!
{
   Weapon.SelectionOrder 1
   Inventory.PickupSound "misc/w_pkup"
   Inventory.PickupMessage "You got the ultimate Legendary C4! straight out of DE_DUST!"
   Weapon.AmmoType "LegendaryC4Ammo"
   Weapon.AmmoGive 1
   Weapon.AmmoUse 0
Weapon.SlotNumber 1
  Weapon.SlotPriority 6.1
   Weapon.Kickback 1000
   Scale 1
   +NOALERT
   +WEAPON.ALT_AMMO_OPTIONAL
   Tag "Legendary C4"
   States
   {
   Spawn:
   	PUC4 A 0 NoDelay ACS_NamedExecuteWithResult("WeaponListCheck",1,1)
  SpawnLoop:
      PUC4 A -1
      Loop
   Ready:
	  TNT1 A 0 A_JumpIfInventory("legendaryC4Ammo",1,1)
	  Goto ReadyUnarmed
	  TNT1 A 0
      BDC4 A 1 A_WeaponReady
	  Loop
   ReadyUnarmed:
      DEC4 C 1 A_WeaponReady
	  Loop
    Deselect:
	TNT1 A 0 A_JumpIfInventory("legendaryC4Ammo",1,1)
	Goto DeselectUnarmed
	TNT1 A 0 A_Lower
	BDC4 A 1 A_Lower
	Goto Deselect+1
   DeselectUnarmed:
      TNT1 A 0 A_Lower
      DEC4 C 1 A_Lower
      Loop
    Select:
	TNT1 A 0 A_JumpIfInventory("legendaryC4Ammo",1,1)
	Goto SelectUnarmed
	TNT1 A 0 A_Raise
	BDC4 A 1 A_Raise
	Goto Select+1
   SelectUnarmed:
      TNT1 A 0 A_Raise
      DEC4 C 1 A_Raise
      Loop
   Fire:
      TNT1 A 0 A_JumpIfInventory("LegendaryC4Ammo",1,1)
	  Goto Ready
	  TNT1 A 0
      BDC4 BCDEFGH 1
	  TNT1 A 0 A_TakeInventory("LegendaryC4Ammo",1)
      BDC4 A 0 A_FireCustomMissile("LegendaryC4Charge",0,1,0,0)
      BDC4 A 0 A_PlayWeaponSound("weapons/pbthrow")
      NULL A 8
	  TNT1 A 0 A_JumpIfInventory("legendaryC4Ammo",1,1)
	  Goto SelectUnarmed
      BDC4 HGFEDCB 1
      BDC4 A 5
      BDC4 A 4 A_ReFire
      Goto Ready
   AltFire:
      TNT1 A 0
	  TNT1 A 0 A_JumpIfInventory("legendaryC4Ammo",1,1)
	  Goto AltFire2
      BDC4 BCDEFGH 1
      NULL AA 4
      DEC4 GFEDC 1
      DEC4 B 0 A_AlertMonsters
      DEC4 B 0 Thing_Projectile(1999,201,0,0,0)
      DEC4 B 0 A_PlayWeaponSound("weapons/pbarm")
      DEC4 B 0 Thing_ReMove(1999)
      DEC4 B 4
      DEC4 A 9
      DEC4 CDEFG 1
      NULL AA 4
      BDC4 HGFEDCB 1
      goto Ready
   AltFire2:
      //TNT1 A 0
      //NULL AA 4
      //DEC4 GFEDC 1
      DEC4 B 0 A_AlertMonsters
      DEC4 B 0 Thing_Projectile(1999,201,0,0,0)
      DEC4 B 0 A_PlayWeaponSound("weapons/pbarm")
      DEC4 B 0 Thing_ReMove(1999)
      DEC4 B 4
      DEC4 A 9
      //DEC4 CDEFG 1
      //NULL AA 4
      goto Ready
   }
}
